288 /* Create the lock file.
289  The lock file will be removed if we catch a signal or when we exit. */
290 /* The old code here used flock on a lock file /etc/mtab~ and deleted
291  this lock file afterwards. However, as rgooch remarks, that has a
292  race: a second mount may be waiting on the lock and proceed as
293  soon as the lock file is deleted by the first mount, and immediately
294  afterwards a third mount comes, creates a new /etc/mtab~, applies
295  flock to that, and also proceeds, so that the second and third mount
296  now both are scribbling in /etc/mtab.
297  The new code uses a link() instead of a creat(), where we proceed
298  only if it was us that created the lock, and hence we always have
299  to delete the lock afterwards. Now the use of flock() is in principle
300  superfluous, but avoids an arbitrary sleep(). */
301 
302 /* Where does the link point to? Obvious choices are mtab and mtab~~.
303  HJLu points out that the latter leads to races. Right now we use
304  mtab~.<pid> instead. Use 20 as upper bound for the length of %d. */
305 #define MOUNTLOCK_LINKTARGET MOUNTED_LOCK "%d"
306 #define MOUNTLOCK_LINKTARGET_LTH (sizeof(MOUNTED_LOCK)+20)
307 
308 void
309 lock_mtab (void) {
310  int tries = 100000, i;
311  char linktargetfile[MOUNTLOCK_LINKTARGET_LTH];
312 
313  at_die = unlock_mtab;
314 
315  if (!signals_have_been_setup) {
316  int sig = 0;
317  struct sigaction sa;
318 
319  sa.sa_handler = handler;
320  sa.sa_flags = 0;
321  sigfillset (&sa.sa_mask);
322 
323  while (sigismember (&sa.sa_mask, ++sig) != -1
324  && sig != SIGCHLD) {
325  if (sig == SIGALRM)
326  sa.sa_handler = setlkw_timeout;
327  else
328  sa.sa_handler = handler;
329  sigaction (sig, &sa, (struct sigaction *) 0);
330  }
331  signals_have_been_setup = 1;
332  }
333 
334  sprintf(linktargetfile, MOUNTLOCK_LINKTARGET, getpid ());
335 
336  i = open (linktargetfile, O_WRONLY|O_CREAT, 0);
337  if (i < 0) {
338  int errsv = errno;
339  /* linktargetfile does not exist (as a file)
340  and we cannot create it. Read-only filesystem?
341  Too many files open in the system?
342  Filesystem full? */
343  die (EX_FILEIO, _("can't create lock file %s: %s "
344  "(use -n flag to override)"),
345  linktargetfile, strerror (errsv));
346  }
347  close(i);
348 
349  /* Repeat until it was us who made the link */
350  while (!we_created_lockfile) {
351  struct flock flock;
352  int errsv, j;
353 
354  j = link(linktargetfile, MOUNTED_LOCK);
355  errsv = errno;
356 
357  if (j == 0)
358  we_created_lockfile = 1;
359 
360  if (j < 0 && errsv != EEXIST) {
361  (void) unlink(linktargetfile);
362  die (EX_FILEIO, _("can't link lock file %s: %s "
363  "(use -n flag to override)"),
364  MOUNTED_LOCK, strerror (errsv));
365  }
366 
367  lockfile_fd = open (MOUNTED_LOCK, O_WRONLY);
368 
369  if (lockfile_fd < 0) {
370  int errsv = errno;
371  /* Strange... Maybe the file was just deleted? */
372  if (errno == ENOENT && tries-- > 0) {
373  if (tries % 200 == 0)
374  usleep(30);
375  continue;
376  }
377  (void) unlink(linktargetfile);
378  die (EX_FILEIO, _("can't open lock file %s: %s "
379  "(use -n flag to override)"),
380  MOUNTED_LOCK, strerror (errsv));
381  }
382 
383  flock.l_type = F_WRLCK;
384  flock.l_whence = SEEK_SET;
385  flock.l_start = 0;
386  flock.l_len = 0;
387 
388  if (j == 0) {
389  /* We made the link. Now claim the lock. */
390  if (fcntl (lockfile_fd, F_SETLK, &flock) == -1) {
391  if (verbose) {
392  int errsv = errno;
393  printf(_("Can't lock lock file %s: %s\n"),
394  MOUNTED_LOCK, strerror (errsv));
395  }
396  /* proceed anyway */
397  }
398  (void) unlink(linktargetfile);
399  } else {
400  static int tries = 0;
401 
402  /* Someone else made the link. Wait. */
403  alarm(LOCK_TIMEOUT);
404  if (fcntl (lockfile_fd, F_SETLKW, &flock) == -1) {
405  int errsv = errno;
406  (void) unlink(linktargetfile);
407  die (EX_FILEIO, _("can't lock lock file %s: %s"),
408  MOUNTED_LOCK, (errno == EINTR) ?
409  _("timed out") : strerror (errsv));
410  }
411  alarm(0);
412  /* Limit the number of iterations - maybe there
413  still is some old /etc/mtab~ */
414  ++tries;
415  if (tries % 200 == 0)
416  usleep(30);
417  if (tries > 100000) {
418  (void) unlink(linktargetfile);
419  close(lockfile_fd);
420  die (EX_FILEIO, _("Cannot create link %s\n"
421  "Perhaps there is a stale lock file?\n"),
422  MOUNTED_LOCK);
423  }
424  close(lockfile_fd);
425  }
426  }
427 }

429 /*
430  * Update the mtab.
431  * Used by umount with null INSTEAD: remove the last DIR entry.
432  * Used by mount upon a remount: update option part,
433  * and complain if a wrong device or type was given.
434  * [Note that often a remount will be a rw remount of /
435  * where there was no entry before, and we'll have to believe
436  * the values given in INSTEAD.]
437  */
438 
439 void
440 update_mtab (const char *dir, nfs_mntent_t *instead) {
441  mntFILE *mfp, *mftmp;
442  const char *fnam = MOUNTED;
443  struct mntentchn mtabhead; /* dummy */
444  struct mntentchn *mc, *mc0, *absent = NULL;
445 
446  if (mtab_does_not_exist() || !mtab_is_writable())
447  return;
448 
449  lock_mtab();
450 
451  /* having locked mtab, read it again */
452  mc0 = mc = &mtabhead;
453  mc->nxt = mc->prev = NULL;
454 
455  mfp = nfs_setmntent(fnam, "r");
456  if (mfp == NULL || mfp->mntent_fp == NULL) {
457  int errsv = errno;
458  nfs_error (_("cannot open %s (%s) - mtab not updated"),
459  fnam, strerror (errsv));
460  goto leave;
461  }
462 
463  read_mntentchn(mfp, fnam, mc);
464 
465  /* find last occurrence of dir */
466  for (mc = mc0->prev; mc && mc != mc0; mc = mc->prev)
467  if (streq(mc->m.mnt_dir, dir))
468  break;
469  if (mc && mc != mc0) {
470  if (instead == NULL) {
471  /* An umount - remove entry */
472  if (mc && mc != mc0) {
473  mc->prev->nxt = mc->nxt;
474  mc->nxt->prev = mc->prev;
475  free(mc);
476  }
477  } else {
478  /* A remount */
479  mc->m.mnt_opts = instead->mnt_opts;
480  }
481  } else if (instead) {
482  /* not found, add a new entry */
483  absent = xmalloc(sizeof(*absent));
484  absent->m = *instead;
485  absent->nxt = mc0;
486  absent->prev = mc0->prev;
487  mc0->prev = absent;
488  if (mc0->nxt == NULL)
489  mc0->nxt = absent;
490  }
491 
492  /* write chain to mtemp */
493  mftmp = nfs_setmntent (MOUNTED_TEMP, "w");
494  if (mftmp == NULL || mftmp->mntent_fp == NULL) {
495  int errsv = errno;
496  nfs_error (_("cannot open %s (%s) - mtab not updated"),
497  MOUNTED_TEMP, strerror (errsv));
498  goto leave;
499  }
500 
501  for (mc = mc0->nxt; mc && mc != mc0; mc = mc->nxt) {
502  if (nfs_addmntent(mftmp, &(mc->m)) == 1) {
503  int errsv = errno;
504  die (EX_FILEIO, _("error writing %s: %s"),
505  MOUNTED_TEMP, strerror (errsv));
506  }
507  }
508 
509  discard_mntentchn(mc0);
510 
511  if (fchmod (fileno (mftmp->mntent_fp),
512  S_IRUSR|S_IWUSR|S_IRGRP|S_IROTH) < 0) {
513  int errsv = errno;
514  fprintf(stderr, _("error changing mode of %s: %s\n"),
515  MOUNTED_TEMP, strerror (errsv));
516  }
517  nfs_endmntent (mftmp);
518 
519  { /*
520  * If mount is setuid and some non-root user mounts sth,
521  * then mtab.tmp might get the group of this user. Copy uid/gid
522  * from the present mtab before renaming.
523  */
524  struct stat sbuf;
525  if (stat (MOUNTED, &sbuf) == 0)
526  chown (MOUNTED_TEMP, sbuf.st_uid, sbuf.st_gid);
527  }
528 
529  /* rename mtemp to mtab */
530  if (rename (MOUNTED_TEMP, MOUNTED) < 0) {
531  int errsv = errno;
532  fprintf(stderr, _("can't rename %s to %s: %s\n"),
533  MOUNTED_TEMP, MOUNTED, strerror(errsv));
534  }
535 
536  leave:
537  unlock_mtab();
538 }
