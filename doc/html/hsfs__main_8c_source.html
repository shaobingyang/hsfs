174 /* Try to build a canonical options string. */
175 static char * hsi_fix_opts_string (int flags, const char *extra_opts) {
176  const struct opt_map *om;
177  char *new_opts;
178 
179  new_opts = xstrdup((flags & MS_RDONLY) ? "ro" : "rw");
180  if (flags & MS_USER) {
181  struct passwd *pw = getpwuid(getuid());
182  if(pw)
183  new_opts = xstrconcat3(new_opts, ",user=", pw->pw_name);
184  }
185 
186  for (om = opt_map; om->nfs_opt != NULL; om++) {
187  if (om->skip)
188  continue;
189  if (om->inv || !om->mask || (flags & om->mask) != om->mask)
190  continue;
191  new_opts = xstrconcat3(new_opts, ",", om->nfs_opt);
192  flags &= ~om->mask;
193  }
194  if (extra_opts && *extra_opts) {
195  new_opts = xstrconcat3(new_opts, ",", extra_opts);
196  }
197 
198  return new_opts;
199 }

279 static void hsi_parse_opt(const char *opt, int *flags, struct fuse_args *args,
280  char *extra_opts, size_t len)
281 {
282  const struct opt_map *om = NULL;
283 
284  for (om = opt_map; om->nfs_opt != NULL; om++) {
285  if (!strcmp (opt, om->nfs_opt)) {
286  if (om->fuse_opt) {
287  hsi_fuse_add_opt(args, om->fuse_opt);
288  if (om->inv)
289  *flags &= ~om->mask;
290  else
291  *flags |= om->mask;
292  } else {
293  WARNING("Not supported opt: %s.", opt);
294  }
295  return;
296  } else if (fuse_lowlevel_is_lib_option(opt)) {
297  hsi_fuse_add_opt(args, opt);
298  return;
299  }
300  }
301 
302  len -= strlen(extra_opts);
303 
304  if (*extra_opts && --len > 0)
305  strcat(extra_opts, ",");
306 
307  if ((len -= strlen(opt)) > 0)
308  strcat(extra_opts, opt);
309 }

312 static void hsi_parse_opts(const char *options, int *flags,
313  struct fuse_args *args, char **udata)
314 {
315  if (options != NULL) {
316  char *opts = xstrdup(options);
317  char *opt = NULL, *p = NULL;
318  size_t len = strlen(opts) + 1; /* include room for a null */
319  int open_quote = 0;
320 
321  *udata = xmalloc(len);
322  **udata = '\0';
323 
324  for (p = opts, opt = NULL; p && *p; p++) {
325  if (!opt)
326  opt = p; /* begin of the option item */
327  if (*p == '"')
328  open_quote ^= 1; /* reverse the status */
329  if (open_quote)
330  continue; /* still in a quoted block */
331  if (*p == ',')
332  *p = '\0'; /* terminate the option item */
333 
334  /* end of option item or last item */
335  if (*p == '\0' || *(p + 1) == '\0') {
336  hsi_parse_opt(opt, flags, args, *udata, len);
337  opt = NULL;
338  }
339  }
340  free(opts);
341  }
342 }

344 static int hsi_parse_cmdline(int argc, char **argv, int *flags,
345  struct fuse_args *args, char **udata)
346 {
347  char *tdata = NULL;
348  int c = 0, ret = 0;
349 
350  while((c = getopt_long(argc, argv, "rvVwfno:h",
351  hsfs_opts, NULL)) != -1) {
352  switch(c) {
353  case 'r':
354  *flags |= MS_RDONLY;
355  break;
356  case 'v':
357  ++verbose;
358  break;
359  case 'V':
360  hsi_print_version();
361  break;
362  case 'w':
363  *flags &= ~MS_RDONLY;
364  break;
365  case 'f':
366  fg = 1;
367  break;
368  case 'n':
369  nomtab = 1;
370  break;
371  case 'o':
372  tdata = optarg;
373  break;
374  case 'h':
375  default:
376  hsi_mount_usage();
377  break;
378 
379  }
380  }
381 
382  if (optind != argc)
383  hsi_mount_usage();
384 
385  /* add subtype args */
386  {
387  char subtype_opt[PAGE_SIZE] = {};
388 
389  sprintf(subtype_opt, "-osubtype=%s", progname);
390  ret = fuse_opt_add_arg(args, subtype_opt);
391  if (ret)
392  goto out;
393  }
394 
395  /* add fs mode */
396  {
397  char *opt = NULL;
398  opt = (*flags) & MS_RDONLY ? "ro" : "rw";
399  ret = hsi_fuse_add_opt(args, opt);
400  if (ret)
401  goto out;
402  }
403 
404  hsi_parse_opts(tdata, flags, args, udata);
405 
406  ret = fuse_daemonize(fg);
407 out:
408  return ret;
409 }

411 int main(int argc, char **argv)
412 {
413  struct fuse_args args = {};
414  struct fuse_chan *ch = NULL;
415  struct fuse_session *se = NULL;
416  char *mountpoint = NULL, *mountspec = NULL, *udata = NULL;
417  struct hsfs_super super = {};
418  int err = -1;
419 
420  progname = basename(argv[0]);
421 
422  if (argv[1] && argv[1][0] == '-') {
423  if(argv[1][1] == 'V')
424  hsi_print_version();
425  else
426  hsi_mount_usage();
427  }
428 
429  if (argc < 3)
430  hsi_mount_usage();
431 
432  mountspec = argv[1];
433  mountpoint = argv[2];
434 
435  argv[2] = argv[0]; /* so that getopt error messages are correct */
436  err = hsi_parse_cmdline(argc - 2, argv + 2, &super.flags,
437  &args, &udata);
438  if (err != 0)
439  goto out;
440 
441  ch = hsx_fuse_mount(mountspec, mountpoint, &args, udata, &super);
442  if (ch == NULL)
443  goto out;
444 
445  if (getuid() == 0)
446  hsi_add_mtab(mountspec, mountpoint, HSFS_TYPE, super.flags,
447  udata);
448 
449  se = fuse_lowlevel_new(&args, &hsfs_oper, sizeof(hsfs_oper), &super);
450  if (se != NULL) {
451  if (fuse_set_signal_handlers(se) != -1) {
452  fuse_session_add_chan(se, ch);
453  err = fuse_session_loop(se);
454  fuse_remove_signal_handlers(se);
455  fuse_session_remove_chan(ch);
456  }
457  fuse_session_destroy(se);
458  }
459 
460  if (getuid() == 0)
461  hsi_del_mtab(mountpoint);
462 
463  hsx_fuse_unmount(mountspec, mountpoint, ch, &super);
464 out:
465  if (udata)
466  free(udata);
467 
468  fuse_opt_free_args(&args);
469 
470  return err ? 1 : 0;
471 }
